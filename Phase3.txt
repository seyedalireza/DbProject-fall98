create table AirPort
(
    ID      int not null,
    City    char(250),
    State   char(250),
    Address char(250),
    Kind    char(250),
    primary key (ID),
    CHECK ( Kind = 'International' or Kind = 'National' or (Kind is null))
);

create table Flight
(
    ID               int not null,
    DateAndTime      date,
    Source           int,
    Dest1            int not null,
    Dest1ArrivalTime date,
    Dest2            int,
    Dest2ArrivalTime date,
    Plane            int,
    PRIMARY KEY (ID),
    FOREIGN KEY (Source) REFERENCES AirPort (ID)
        on update cascade
        on delete cascade,
    FOREIGN KEY (Dest1) REFERENCES AirPort (ID)
        on update cascade
        on delete cascade,
    FOREIGN KEY (Dest2) REFERENCES AirPort (ID)
        on update cascade
        on delete cascade,
    FOREIGN KEY (Plane) REFERENCES Plane (ID)
        on update cascade
        on delete cascade

);

(
    AirAgencyID int not null,

        on delete cascade,
    FOREIGN KEY (FlightID, ClassID) REFERENCES FlightClass (FlightID, ID)
